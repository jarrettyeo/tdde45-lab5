15 # The type of this value might be Bool.
16 # However, to enable a limited amount of back-propagagation,
17 # we also keep some information about how this Bool value was created.
18 # In particular, if you branch on this value, then may assume that in
19 # the true branch, the type of `var` will be limited by `vtype` and in
20 # the false branch, it will be limited by `elsetype`. Example:
21 # ```
22 # cond = isa(x::Union{Int, Float}, Int)::Conditional(x, Int, Float)
23 # if cond
24 # # May assume x is `Int` now
25 # else
26 # # May assume x is `Float` now
27 # end
28 # ```

30 var::Slot
31 vtype
32 elsetype

36 @nospecialize(nottype))

38 end

182 a === b && return true
